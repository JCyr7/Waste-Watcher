export default class WasteEntry extends Component {

  // method to conditionally display different numbers of days for each month
  dayOption(month) {
    if (month === 'February') {
      return dayDropdown.slice(0, 28)
    } else if (
      month === 'April' ||
      month === 'June' ||
      month === 'September' ||
      month === 'November'
    ) {
      return dayDropdown.slice(0, 30)
    } else {
      return dayDropdown
    }
  }
